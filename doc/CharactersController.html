def create
  @errors = {"error" => Array.new, "warning" => Array.new}
  # Get account data so we can associate it with the character-to-be-created
  # Check if Some kind of possible API-data is entered
  user_id, api_key = params['character']['user_id'], params['character']['api_key']
  character_ids = params['character']['character_ids'];
  begin
    @characters = Character.find_by_api!(user_id, api_key)
  rescue Exception => e
    @errors["error"] << e.message
    logger.warn e.backtrace.to_yaml
    render "create_failed"
    return
  end
  
  # If user has choosen his characters, check for legitimacy
  # (prevent forging character names) and try to register those characters
  if character_ids.blank?
    # if there are characters found for his API data, but he has not chosen
    # any as of now, display character selection boxes
    render "expand_characters"
    return
  else 
    # Register characters
    @characters = @characters.select{ |c| character_ids.include? c[:character_id]}
    if @characters.blank?
      # Display error if user tried to inject characters that are not his into the form
      @errors["error"] = "Selected character names do not match characters from the API"
    else 
      @characters.each do |c|
        # Go through each character and add it to his account
        if @character = current_account.characters.create(c)
          current_account.update_attributes(:main_character => @character.name)
        else
          @errors["error"] = "Registration of character #{self.name} failed."
        end
      end
    end
    
    # If there have been errors, display those. Else, redirect user to ???
    unless @errors.any?
      render "create_failed"
    else
      render 'create'
    end
  end
end
