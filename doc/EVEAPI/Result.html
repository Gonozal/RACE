def self.new(xml)
	@xml = Nokogiri::XML(xml)
	# handle possible API errors
	if @xml.xpath("/eveapi/error").length > 0
		# API error detected
		error = @xml.xpath("/eveapi/error").first
		raise EVEAPI::Exception::APIError.new(error.text, error['code'])
	elsif @xml.xpath("/eveapi/result").length < 1
		raise EVEAPI::Exception::APIError.new("Unknown API Error detected. Result element not found.")
	else
		# no errors detected, continuing with parsing
		node = Node.new @xml.xpath "/eveapi/result"
		node.current_time = @xml.at_xpath("/eveapi/currentTime").content
		node.cached_until = @xml.at_xpath("/eveapi/cachedUntil").content
		return node
	end
end
